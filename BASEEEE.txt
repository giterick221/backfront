CREATE TABLE usuarios(
    id_usuario varchar(20) NOT NULL,
    pass_usuario varchar(64),
    nombre_completo_usuario varchar (50) NOT NULL,
    telefono_usuario varchar(20) DEFAULT "No agregado",
    oculto_usuario char(1) DEFAULT "0",
    fecha_creacion_usuario timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_usuario)
    );

CREATE TABLE tipo_usuario(
    id_tipo_usuario int NOT NULL AUTO_INCREMENT,
    tipo_usuario varchar(10) DEFAULT "fiscal",
    id_usuario_tipo varchar (20),
    PRIMARY KEY (id_tipo_usuario),
    FOREIGN KEY (id_usuario_tipo) REFERENCES usuarios(id_usuario)
    );

CREATE TABLE conminatorias (
    id_conmi int NOT NULL AUTO_INCREMENT,
    cud_conmi varchar(25),
    nurej_conmi varchar(25),
    fiscal_conmi varchar(50),
    denunciado_conmi varchar(255),
    denunciante_conmi varchar(255),
    delito_conmi varchar(50),
    fecha_notif_cd_conmi date,
    fecha_notif_fiscal_conmi date,
    tiempo_restante_conmi date,
    realizado_conmi char(1),
    oculto_conmi char(1) DEFAULT '0',
    fecha_creacion_conmi timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
    id_usuario_conmi varchar (20),
    PRIMARY KEY (id_conmi),
    FOREIGN KEY (id_usuario_conmi) REFERENCES usuarios(id_usuario)
);
